Backbone Complety

Simple Backbone based replacement for long and boring select boxes. It supports searching, multiple selection and custom templates for search results.

Basic Autocomplete

Autocomplete for Text Input

Autocomplete for Textarea

var complety = new Backbone.Complety({
  //search source
  collection: famousCarBrands,
  //div selector where widget will be rendered
  targetContainer: '.input-complete',
  //attribute of the model where to search
  searchAttr: "brand"
});
var completyArea = new Backbone.Complety({
  collection: famousCarBrands,
  targetContainer: '.area-complete',

  //Set to true if you want textarea to be
  //rendered.
  isArea: true,

  searchAttr: "brand"
});

Custom Template for Autocomplete

Multiple Autocomplete for Text Input

Multiple Autocomplete for Textarea

var complety = new Backbone.Complety({
  collection: source to look,
  targetContainer: selector for div,
  searchAttr: attribute to search,
  template: {
    template: Template,
    tmpltData: {
      keys : model attribute
    }
});
var customCompletyArea = new Backbone.Complety({
  collection: famousCarBrands,
  targetContainer: '.custom-area-complete',
  isArea: true,
  isMultiple: true,
  searchAttr: "brand",
  ignoreCase: true,
  Template: {
    template: '<table><tr class="trow"><td>
            <img height="40" src="<%= img %>"></td>
            <td class="title"><%= name %></td></tr>
            </table>',
    values: {
      name: 'brand',
      img: 'imgURL'
    }
  }
});
